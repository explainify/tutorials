let t = 0;
function setup(){ createCanvas(280, 160); }
function draw(){
  background(15);
  stroke(96,165,250);
  noFill();
  beginShape();
  for (let x=0; x<width; x++){
    let y = 0.02*(x-140)*(x-140) + 20 + 10*sin(t);
    vertex(x, y);
  }
  endShape();
  t += 0.03;
}
