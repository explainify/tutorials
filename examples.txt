::: accordion
## What is event-level data?
Each row is one **hospital admission**. The same patient can appear many times.

## What is person-level analysis?
You count each **person** at most once, so heavy users don't dominate.

## When do I use which?
Use **event-level** for workload; use **person-level** for prevalence.
:::

::: html(id="counter-demo")
<div class="counter-wrap" style="display:flex;align-items:center;gap:0.75rem;flex-wrap:wrap;">
  <button type="button" id="counter-minus" style="padding:0.4rem 0.8rem;font-size:1.1rem;">−</button>
  <span id="counter-value" style="min-width:2rem;text-align:center;font-weight:bold;">0</span>

:::

::: js(show_code=false, silent=true)
(function () {
  var el = document.getElementById("counter-value");
  var minus = document.getElementById("counter-minus");
  var plus = document.getElementById("counter-plus");
  if (!el || !minus || !plus) return;
  var n = 0;
  function update() { el.textContent = n; }
  minus.addEventListener("click", function () { n = Math.max(0, n - 1); update(); });
  plus.addEventListener("click", function () { n += 1; update(); });
  update();
})();
:::
